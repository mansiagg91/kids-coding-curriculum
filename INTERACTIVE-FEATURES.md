import neopixel
import time

def rainbow_show():
    """Beautiful rainbow animation on LED strip"""
    colors = [
        (255, 0, 0),    # Red
        (255, 127, 0),  # Orange  
        (255, 255, 0),  # Yellow
        (0, 255, 0),    # Green
        (0, 0, 255),    # Blue
        (75, 0, 130),   # Indigo
        (148, 0, 211)   # Violet
